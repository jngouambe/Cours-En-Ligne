using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Globalization;


namespace Bibliothèque
{
    class Program
    {

        //Déclaration de la structure d'un livre.
        struct Livre
        {
            private string isbn;
            private string auteur;
            private string titre;
            private string editeur;
            private ushort nbrPages;
            private double prixVente;

            //Création de la proprité ISBN.
            public string ISBN
            {
                get
                {
                    return isbn;
                }
                set
                {
                    long chiffre;
                    if (value.Length != 13 || !long.TryParse(value, out chiffre))
                    {
                        throw new Exception($" Un code ISBN doit être composé de 13 caractères numériques.");
                    }
                    else
                    {
                        isbn = value;
                    }
                }
            }
            public string Auteur
            {
                get
                {
                    return auteur;
                }
                set
                {
                    auteur = value;
                }
            }
            public string Titre
            {
                get
                {
                    return titre;
                }
                set
                {
                    titre = value;
                }
            }
            public string Editeur
            {
                get
                {
                    return editeur;
                }
                set
                {
                    editeur = value;
                }
            }

            public string NombrePages
            {
                get
                {
                    return nbrPages.ToString();
                }
                set
                {
                    ushort nombre;
                    if (!ushort.TryParse(value, out nombre))
                    {
                        throw new Exception($" Le nombre de pages d'un livre doit être un nombre entier positif.");
                    }
                    else
                    {
                        nbrPages = nombre;
                    }
                }
            }

            public string PrixVente
            {
                get
                {
                    return CultureInfo.CurrentCulture.NumberFormat.CurrencySymbol + " " + prixVente.ToString("0.00");
                }
                set
                {
                    double nombre;
                    if (CultureInfo.CurrentCulture.NumberFormat.CurrencyDecimalSeparator == ",")
                    {
                        value = value.Replace(".", ",");
                    }
                    else
                    {
                        value = value.Replace(",", ".");
                    }
                    if (!double.TryParse(value, out nombre))
                    {
                        throw new Exception(" Le prix de vente d'un livre doit être une valeur monétaire positive.");
                    }
                    else
                    {
                        if (nombre <= 0)
                        {
                            throw new Exception(" Le prix de vente d'un livre doit être une valeur monétaire positive");
                        }
                        else
                        {
                            prixVente = nombre;
                        }
                    }

                }
            }

            public void Augmentation(double valeurAugmentation)
            {
                prixVente = prixVente * (1 + valeurAugmentation / 100);
            }

        }


        //Méthode principale de l'application.
        static void Main(string[] args)
        {
            //Création des variables utilisées dans le programme.
            Livre livre = new Livre();
            bool reponse;

            Console.WriteLine();
            //Boucle de contrôle de la saisie du code ISBN d'un livre.
            do
            {
                try
                {
                    //Saisie d'un code ISBN d'un livre.
                    Console.Write(" Donnez-moi le code ISBN du livre: ");
                    livre.ISBN = Console.ReadLine();
                    reponse = true;
                }
                //Si la saisie n'est pas valide, affichage du message d'erreur.
                catch (Exception e)
                {
                    Console.WriteLine();
                    Console.WriteLine(e.Message);
                    Console.WriteLine();
                    reponse = false;
                }

            } while (!reponse);

            reponse = false;

            //Saisie du titre, du nom de l'auteur et de l'éditeur du livre.
            Console.Write(" Donnez-moi le titre du livre: ");
            livre.Titre = Console.ReadLine();
            Console.Write(" Donnez-moi le nom de l'auteur du livre: ");
            livre.Auteur = Console.ReadLine();
            Console.Write(" Donnez-moi le nom de l'éditeur du livre: ");
            livre.Editeur = Console.ReadLine();

            //Boucle de contrôle de la saisie du nombre de pages du livre.
            do
            {
                try
                {
                    Console.Write(" Donnez-moi le nombre de pages que contient le livre: ");
                    //Saisie du nombre de page du livre.
                    livre.NombrePages = Console.ReadLine();
                    reponse = true;
                }
                //Si la saisie n'est pas valide, affichage du message d'erreur.
                catch (Exception e)
                {
                    Console.WriteLine();
                    Console.WriteLine(e.Message);
                    Console.WriteLine();
                    reponse = false;
                }

            } while (!reponse);

            reponse = false;
            //Boucle de contrôle de la saisie du prix de vente du livre.
            do
            {
                try
                {
                    Console.Write(" Donnez-moi le prix de vente du livre: ");
                    //Saisie du prix de vente du livre.
                    livre.PrixVente = Console.ReadLine();
                    reponse = true;

                }
                catch (Exception e)
                {
                    Console.WriteLine();
                    Console.WriteLine(e.Message);
                    Console.WriteLine();
                    reponse = false;
                }
            } while (!reponse);

            //Affichage des informations saisies sur le livre.
            Console.WriteLine();
            Console.WriteLine($" ISBN: {livre.ISBN}");
            Console.WriteLine($" Titre: {livre.Titre}");
            Console.WriteLine($" Auteur: {livre.Auteur}");
            Console.WriteLine($" Éditeur: {livre.Editeur}");
            Console.WriteLine($" Nombre de pages: {livre.NombrePages}");
            Console.WriteLine($" Prix de vente: {livre.PrixVente}");
            reponse = false;
            do
            {
                //Déclaratio de la variable qui sera utilidée pour récupérer la réponse de l'utilisateur.
                char augmentation;
                Console.WriteLine();
                Console.Write(" Désirez-vous augmenter le prix de vente du livre ? (O ou N) ");
                //Récupération de la réponse de l'Utilisateur.
                reponse = char.TryParse(Console.ReadLine().ToUpper(), out augmentation);
                //Si le transtypage a réussi.
                if (reponse)
                {
                    //Si la réponse est différente de 'O' ou 'N'.
                    if (augmentation != 'O' && augmentation != 'N')
                    {
                        Console.WriteLine();
                        Console.WriteLine("Vous devez répondre par O ou N.");
                        Console.WriteLine();
                        reponse = false;
                    }
                    //Si la réponse est soit 'O' ou 'N'.
                    else
                    {
                        reponse = true;
                        //Si la réponse est égale à 'O'.
                        if (augmentation == 'O')
                        {
                            //Appel de la méthode AugmenterPrix() avec l'objet livre passé en paramètre.
                            AugmenterPrix(livre);
                        }
                    }
                }
                else
                {
                    Console.WriteLine();
                    Console.WriteLine(" Vous devez répondre par O ou N.");
                    Console.WriteLine();
                    reponse = false;
                }
            } while (!reponse);
           
            Console.ReadKey();
        }
        static void AugmenterPrix(Livre livre)
        {
            //Déclaration des variables utilisées par la méthode.
            bool valide;
            double augmentation;
            string valeur;
            //Message à l'utilisateur.
            Console.WriteLine();
            Console.WriteLine(" Donnez-moi la valeur de l'augmentation que vous désirez appliquer");
            Console.WriteLine(" au prix de vente du livre.(Exemple, 10 pour 10%): ");
            //Saisie de la valeur du taux d'augmentation.
            valeur = Console.ReadLine();
            //Vérification du format de la saisie.
            if (CultureInfo.CurrentCulture.NumberFormat.CurrencyDecimalSeparator == ",")
            {
                valeur = valeur.Replace(".", ",");
            }
            else
            {
                valeur = valeur.Replace(",", ".");
            }
            //Conversion de la saisie en valeur de type double.
            valide = double.TryParse(Console.ReadLine(), out augmentation);
            if (valide && augmentation > 0)
            {
                livre.Augmentation(augmentation);
                //Affichage des informations saisies sur le livre après l'augmentation.
                Console.WriteLine();
                Console.WriteLine($" Après une augmentation de {augmentation}% du prix de vente,");
                Console.WriteLine(" les informations sur le livre sont maintenant les suivantes.");
                Console.WriteLine();
                Console.WriteLine($" ISBN: {livre.ISBN}");
                Console.WriteLine($" Titre: {livre.Titre}");
                Console.WriteLine($" Auteur: {livre.Auteur}");
                Console.WriteLine($" Éditeur: {livre.Editeur}");
                Console.WriteLine($" Nombre de pages: {livre.NombrePages}");
                Console.WriteLine($" Prix de vente: {livre.PrixVente}");
            }
            //Si la conversion n'a pas réussi ou que la valeur de l'augmentation est égale ou inférieur à 0.
            else
            {
                Console.WriteLine();
                Console.WriteLine(" La valeur saisie n'est pas une valeur numérique positive valide.");
                Console.WriteLine(" Impossible d'effectuer l'opéraion d'augmentation de prix demandée.");
            }
        }
    }

}
